```typescript
"use client";

import { useState, useEffect, useCallback } from "react";
import { getSubtaskProgress } from "@/server/actions/subtasks";

interface SubtaskProgressStory {
  id: string;
  code: string;
  title: string;
  status: string;
  subtasksCreated: number;
  durationMs: number | null;
  errorMsg: string | null;
}

interface SubtaskProgress {
  runId: string;
  status: string;
  phase: string;
  phaseDetail: string | null;
  totalStories: number;
  completedStories: number;
  failedStories: number;
  skippedStories: number;
  totalSubtasks: number;
  progress: number;
  estimatedRemainingMs: number;
  stories: SubtaskProgressStory[];
  startedAt: Date | null;
  completedAt: Date | null;
  logs: string | null;
}

interface UseSubtaskProgressOptions {
  onComplete?: () => void;
  onError?: (error: string) => void;
}

export function useSubtaskProgress(
  runId: string | null,
  options: UseSubtaskProgressOptions = {}
) {
  const [progress, setProgress] = useState<SubtaskProgress | null>(null);
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const fetchProgress = useCallback(async () => {
    if (!runId) return;

    try {
      const data = await getSubtaskProgress(runId);
      if (data) {
        setProgress(data as SubtaskProgress);

        // Check for completion
        if (data.status === "SUCCEEDED" || data.status === "PARTIAL") {
          options.onComplete?.();
        } else if (data.status === "FAILED") {
          options.onError?.("Subtask generation failed");
        }
      }
    } catch (err) {
      const message = err instanceof Error ? err.message : "Failed to fetch progress";
      setError(message);
      options.onError?.(message);
    }
  }, [runId, options]);

  useEffect(() => {
    if (!runId) {
      setProgress(null);
      return;
    }

    setIsLoading(true);
    fetchProgress().finally(() => setIsLoading(false));

    // Poll every second while running
    const interval = setInterval(() => {
      if (progress?.status === "RUNNING" || progress?.status === "QUEUED") {
        fetchProgress();
      }
    }, 1000);

    return () => clearInterval(interval);
  }, [runId, fetchProgress, progress?.status]);

  return { progress, isLoading, error, refetch: fetchProgress };
}
```

```typescript
"use client";

import { useState } from "react";
import { useRouter } from "next/navigation";
import {
  Dialog,
  DialogContent,
  DialogDescription,
  DialogHeader,
  DialogTitle,
  DialogTrigger,
} from "@/components/ui/dialog";
import { Button } from "@/components/ui/button";
import { Label } from "@/components/ui/label";
import { RadioGroup, RadioGroupItem } from "@/components/ui/radio-group";
import { Checkbox } from "@/components/ui/checkbox";
import { Badge } from "@/components/ui/badge";
import { ListTodo, Loader2 } from "lucide-react";
import { startGenerateSubtasks, type SubtaskGenerationOptions } from "@/server/actions/subtasks";
import { toast } from "sonner";

interface Story {
  id: string;
  code: string;
  title: string;
  _count?: { subtasks: number };
}

interface SubtaskConfigDialogProps {
  epicId: string;
  stories: Story[];
  onStarted?: (runId: string) => void;
}

export function SubtaskConfigDialog({ epicId, stories, onStarted }: SubtaskConfigDialogProps) {
  const router = useRouter();
  const [open, setOpen] = useState(false);
  const [isLoading, setIsLoading] = useState(false);

  // Step 1: Story selection
  const [selectedStoryIds, setSelectedStoryIds] = useState<string[]>(
    stories.map((s) => s.id)
  );
  const [existingBehavior, setExistingBehavior] = useState<"skip" | "replace">("skip");

  // Step 2: Mode selection
  const [mode, setMode] = useState<"compact" | "standard" | "detailed">("standard");
  const [pacing, setPacing] = useState<"safe" | "normal" | "fast">("normal");

  const storiesWithSubtasks = stories.filter((s) => (s._count?.subtasks || 0) > 0);
  const selectedStories = stories.filter((s) => selectedStoryIds.includes(s.id));

  const handleToggleStory = (storyId: string) => {
    setSelectedStoryIds((prev) =>
      prev.includes(storyId)
        ? prev.filter((id) => id !== storyId)
        : [...prev, storyId]
    );
  };

  const handleSelectAll = () => {
    setSelectedStoryIds(stories.map((s) => s.id));
  };

  const handleStart = async () => {
    if (selectedStoryIds.length === 0) {
      toast.error("Please select at least one story");
      return;
    }

    setIsLoading(true);
    try {
      const options: SubtaskGenerationOptions = {
        mode,
        existingBehavior,
        pacing,
      };

      const result = await startGenerateSubtasks(
        epicId,
        selectedStoryIds.length === stories.length ? null : selectedStoryIds,
        options
      );

      if (result.success && result.runId) {
        toast.success(`Started generating subtasks for ${result.storyCount} stories`);
        setOpen(false);
        onStarted?.(result.runId);
        router.refresh();
      } else {
        toast.error(result.error || "Failed to start generation");
      }
    } catch (error) {
      toast.error("Failed to start subtask generation");
    } finally {
      setIsLoading(false);
    }
  };

  const modeDescriptions = {
    compact: "3-5 subtasks per story, core tasks only",
    standard: "5-8 subtasks per story, includes testing tasks",
    detailed: "8-12 subtasks per story, comprehensive breakdown",
  };

  return (
    <Dialog open={open} onOpenChange={setOpen}>
      <DialogTrigger asChild>
        <Button variant="outline" size="sm" disabled={stories.length === 0}>
          <ListTodo className="h-4 w-4 mr-2" />
          Generate Subtasks
        </Button>
      </DialogTrigger>
      <DialogContent className="max-w-lg">
        <DialogHeader>
          <DialogTitle>Generate Subtasks</DialogTitle>
          <DialogDescription>
            Break down user stories into implementable subtasks.
          </DialogDescription>
        </DialogHeader>

        <div className="space-y-6 py-4">
          {/* Story Selection */}
          <div className="space-y-3">
            <div className="flex items-center justify-between">
              <Label>Stories to process</Label>
              <Button
                variant="ghost"
                size="sm"
                onClick={handleSelectAll}
                className="h-auto py-1 px-2 text-xs"
              >
                Select All ({stories.length})
              </Button>
            </div>
            <div className="max-h-40 overflow-y-auto space-y-2 border rounded-md p-2">
              {stories.map((story) => (
                <div key={story.id} className="flex items-center space-x-2">
                  <Checkbox
                    id={story.id}
                    checked={selectedStoryIds.includes(story.id)}
                    onCheckedChange={() => handleToggleStory(story.id)}
                  />
                  <label
                    htmlFor={story.id}
                    className="flex-1 text-sm cursor-pointer"
                  >
                    <Badge variant="outline" className="mr-2">
                      {story.code}
                    </Badge>
                    {story.title}
                    {(story._count?.subtasks || 0) > 0 && (
                      <span className="ml-2 text-xs text-muted-foreground">
                        ({story._count?.subtasks} existing)
                      </span>
                    )}
                  </label>
                </div>
              ))}
            </div>
          </div>

          {/* Existing Subtasks Behavior */}
          {storiesWithSubtasks.length > 0 && (
            <div className="space-y-3">
              <Label>For stories with existing subtasks</Label>
              <RadioGroup
                value={existingBehavior}
                onValueChange={(v) => setExistingBehavior(v as "skip" | "replace")}
              >
                <div className="flex items-center space-x-2">
                  <RadioGroupItem value="skip" id="skip" />
                  <label htmlFor="skip" className="text-sm">
                    Skip (keep existing subtasks)
                  </label>
                </div>
                <div className="flex items-center space-x-2">
                  <RadioGroupItem value="replace" id="replace" />
                  <label htmlFor="replace" className="text-sm">
                    Replace (delete and regenerate)
                  </label>
                </div>
              </RadioGroup>
            </div>
          )}

          {/* Mode Selection */}
          <div className="space-y-3">
            <Label>Generation mode</Label>
            <RadioGroup
              value={mode}
              onValueChange={(v) => setMode(v as typeof mode)}
            >
              {(["compact", "standard", "detailed"] as const).map((m) => (
                <div key={m} className="flex items-center space-x-2">
                  <RadioGroupItem value={m} id={m} />
                  <label htmlFor={m} className="text-sm">
                    <span className="font-medium capitalize">{m}</span>
                    <span className="text-muted-foreground ml-2">
                      ({modeDescriptions[m]})
                    </span>
                  </label>
                </div>
              ))}
            </RadioGroup>
          </div>
        </div>

        <div className="flex justify-between items-center pt-4 border-t">
          <p className="text-sm text-muted-foreground">
            {selectedStories.length} stories selected
          </p>
          <div className="flex gap-2">
            <Button variant="outline" onClick={() => setOpen(false)}>
              Cancel
            </Button>
            <Button onClick={handleStart} disabled={isLoading || selectedStoryIds.length === 0}>
              {isLoading && <Loader2 className="h-4 w-4 mr-2 animate-spin" />}
              Start Generation
            </Button>
          </div>
        </div>
      </DialogContent>
    </Dialog>
  );
}
```

Create progress display following batch-story-run-progress.tsx pattern but simplified:

```typescript
"use client";

import { useSubtaskProgress } from "@/hooks/use-subtask-progress";
import { Card, CardContent, CardDescription, CardHeader, CardTitle } from "@/components/ui/card";
import { Progress } from "@/components/ui/progress";
import { Badge } from "@/components/ui/badge";
import { Button } from "@/components/ui/button";
import { ScrollArea } from "@/components/ui/scroll-area";
import { CheckCircle2, XCircle, Loader2, Clock, ListTodo } from "lucide-react";
import { cancelSubtaskGeneration } from "@/server/actions/subtasks";
import { toast } from "sonner";

interface SubtaskRunProgressProps {
  runId: string;
  onComplete?: () => void;
  onClose?: () => void;
}

export function SubtaskRunProgress({ runId, onComplete, onClose }: SubtaskRunProgressProps) {
  const { progress, isLoading, error } = useSubtaskProgress(runId, { onComplete });

  const handleCancel = async () => {
    const result = await cancelSubtaskGeneration(runId);
    if (result.success) {
      toast.success("Generation cancelled");
    } else {
      toast.error(result.error || "Failed to cancel");
    }
  };

  const formatDuration = (ms: number) => {
    const seconds = Math.floor(ms / 1000);
    const minutes = Math.floor(seconds / 60);
    if (minutes > 0) {
      return `${minutes}m ${seconds % 60}s`;
    }
    return `${seconds}s`;
  };

  const getStatusIcon = (status: string) => {
    switch (status) {
      case "COMPLETED":
        return <CheckCircle2 className="h-4 w-4 text-green-500" />;
      case "FAILED":
        return <XCircle className="h-4 w-4 text-red-500" />;
      case "GENERATING":
      case "SAVING":
        return <Loader2 className="h-4 w-4 text-blue-500 animate-spin" />;
      case "SKIPPED":
        return <Clock className="h-4 w-4 text-gray-400" />;
      default:
        return <Clock className="h-4 w-4 text-gray-400" />;
    }
  };

  if (isLoading && !progress) {
    return (
      <Card>
        <CardContent className="flex items-center justify-center py-8">
          <Loader2 className="h-6 w-6 animate-spin" />
        </CardContent>
      </Card>
    );
  }

  if (error || !progress) {
    return (
      <Card>
        <CardContent className="py-8 text-center text-muted-foreground">
          {error || "No progress data available"}
        </CardContent>
      </Card>
    );
  }

  const isRunning = progress.status === "RUNNING" || progress.status === "QUEUED";
  const isComplete = progress.status === "SUCCEEDED" || progress.status === "PARTIAL" || progress.status === "FAILED";

  return (
    <Card>
      <CardHeader className="pb-2">
        <div className="flex items-center justify-between">
          <div>
            <CardTitle className="flex items-center gap-2">
              <ListTodo className="h-5 w-5" />
              Subtask Generation
            </CardTitle>
            <CardDescription>{progress.phaseDetail || progress.phase}</CardDescription>
          </div>
          <Badge
            variant={
              progress.status === "SUCCEEDED"
                ? "default"
                : progress.status === "FAILED"
                ? "destructive"
                : "secondary"
            }
          >
            {progress.status}
          </Badge>
        </div>
      </CardHeader>
      <CardContent className="space-y-4">
        {/* Progress bar */}
        <div className="space-y-2">
          <div className="flex justify-between text-sm">
            <span>{progress.completedStories} of {progress.totalStories} stories</span>
            <span>{progress.totalSubtasks} subtasks created</span>
          </div>
          <Progress value={progress.progress} />
        </div>

        {/* Estimated time */}
        {isRunning && progress.estimatedRemainingMs > 0 && (
          <p className="text-sm text-muted-foreground">
            Estimated: {formatDuration(progress.estimatedRemainingMs)} remaining
          </p>
        )}

        {/* Story list */}
        <ScrollArea className="h-48">
          <div className="space-y-1">
            {progress.stories.map((story) => (
              <div
                key={story.id}
                className="flex items-center justify-between py-1 px-2 rounded hover:bg-muted/50"
              >
                <div className="flex items-center gap-2">
                  {getStatusIcon(story.status)}
                  <Badge variant="outline" className="text-xs">
                    {story.code}
                  </Badge>
                  <span className="text-sm truncate max-w-[200px]">{story.title}</span>
                </div>
                <div className="flex items-center gap-2 text-xs text-muted-foreground">
                  {story.subtasksCreated > 0 && (
                    <span>+{story.subtasksCreated}</span>
                  )}
                  {story.durationMs && (
                    <span>{formatDuration(story.durationMs)}</span>
                  )}
                  {story.errorMsg && (
                    <span className="text-red-500 truncate max-w-[100px]" title={story.errorMsg}>
                      {story.errorMsg}
                    </span>
                  )}
                </div>
              </div>
            ))}
          </div>
        </ScrollArea>

        {/* Actions */}
        <div className="flex justify-end gap-2 pt-2 border-t">
          {isRunning && (
            <Button variant="outline" size="sm" onClick={handleCancel}>
              Cancel
            </Button>
          )}
          {isComplete && onClose && (
            <Button size="sm" onClick={onClose}>
              Close
            </Button>
          )}
        </div>
      </CardContent>
    </Card>
  );
}
```
